66 void Intake::SetOutput
67 (
68  MechanismControl::MECHANISM_CONTROL_TYPE controlType,
69  double value
70 )
71 {
72  if ( controlType != MechanismControl::MECHANISM_CONTROL_TYPE::PERCENT_OUTPUT )
73  {
74  Logger::GetLogger()->LogError( "Intake::SetOutput", "invalid Control type option" );
75  }
76 
77  // Make sure value is in range (-1.0 to 1.0) and then set the percent output
78  // on the motor (2 calls)
79  if ( m_motor != nullptr )
80  {
81  auto pct = clamp( value, -1.0, 1.0 );
82  m_motor->SetControlMode( IDragonMotorController::DRAGON_CONTROL_MODE::PERCENT_OUTPUT );
83  m_motor->Set( pct );
84  }
85  else
86  {
87  Logger::GetLogger()->LogError( "Intake::SetOutput", "No motors" );
88  }
89 }
