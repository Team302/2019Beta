49 shared_ptr<IDragonMotorController> DragonMotorControllerFactory::CreateMotorController
50 (
51  string mtype,
52  int canID,
53  int pdpID,
54  string usage,
55  bool inverted,
56  bool sensorInverted,
57  ctre::phoenix::motorcontrol::FeedbackDevice feedbackDevice,
58  int countsPerRev,
59  float gearRatio,
60  bool brakeMode,
61  int slaveTo,
62  int peakCurrentDuration,
63  int continuousCurrentLimit,
64  int peakCurrentLimit,
65  bool enableCurrentLimit
66 )
67 {
68  shared_ptr<IDragonMotorController> controller;
69 
70  auto hasError = false;
71 
72  auto type = m_typeMap.find(mtype)->second;
73  if ( type == MOTOR_TYPE::TALONSRX )
74  {
75  // TODO:: set PDP ID
76  auto talon = new DragonTalon(m_usageMap.find(usage)->second, canID, countsPerRev, gearRatio );
77  talon->EnableBrakeMode( brakeMode );
78  talon->Invert( inverted );
79  talon->SetSensorInverted( sensorInverted );
80  talon->ConfigSelectedFeedbackSensor( feedbackDevice, 0, 50 );
81  talon->ConfigSelectedFeedbackSensor( feedbackDevice, 1, 50 );
82 
83  talon->ConfigPeakCurrentLimit( peakCurrentLimit, 50 );
84  talon->ConfigPeakCurrentDuration( peakCurrentDuration, 50 );
85  talon->ConfigContinuousCurrentLimit( continuousCurrentLimit, 50 );
86  talon->EnableCurrentLimiting( enableCurrentLimit );
87 
88  if ( slaveTo > -1 )
89  {
90  talon->SetAsSlave( slaveTo );
91  }
92  controller.reset( talon );
93  }
94  else if ( type == MOTOR_TYPE::BRUSHED_SPARK_MAX || type == MOTOR_TYPE::BRUSHLESS_SPARK_MAX )
95  {
96  auto brushedBrushless = (type == MOTOR_TYPE::BRUSHED_SPARK_MAX) ? CANSparkMax::MotorType::kBrushed : CANSparkMax::MotorType::kBrushless;
97  auto smax = new DragonSparkMax( canID, m_usageMap.find(usage)->second, brushedBrushless, gearRatio );
98  smax->Invert( inverted );
99  smax->EnableBrakeMode( brakeMode );
100  smax->InvertEncoder( sensorInverted );
101  smax->EnableCurrentLimiting( enableCurrentLimit );
102  smax->SetSmartCurrentLimiting( continuousCurrentLimit );
103  if ( slaveTo > -1 )
104  {
105  DragonSparkMax* master = nullptr;
106  if ( GetController( slaveTo ) != nullptr )
107  {
108  master = dynamic_cast<DragonSparkMax*>( GetController( slaveTo ).get() );
109  }
110  if ( master != nullptr )
111  {
112  smax->Follow( master );
113  }
114  else
115  {
116  string msg = "invalid Slave to ID ";
117  msg += to_string( slaveTo );
118  Logger::GetLogger()->LogError( "DragonMotorControllerFactory::CreateMotorController", msg );
119  }
120  }
121  controller.reset( smax );
122  }
123  else
124  {
125  hasError = true;
126  }
127 
128  if ( !hasError )
129  {
130  m_canControllers[ canID ] = controller;
131  }
132  return controller;
133 }
